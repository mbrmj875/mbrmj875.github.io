// نظام تسجيل النقاط
void update_score(GameState_t *gs, int lines_cleared) {
    switch (lines_cleared) {
        case 1: gs->score += 100; break;
        case 2: gs->score += 300; break;
        case 3: gs->score += 700; break;
        case 4: gs->score += 1500; break;
    }
    // تحديث المستوى كل 600 نقطة
    gs->level = (gs->score / 600) + 1;
    gs->speed = 1000 - (gs->level * 100);
}
