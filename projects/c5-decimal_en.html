Practical Usage Examples

Example 1: Precise Financial Calculations
#include "s21_decimal.h" void financial_calculations() {   s21_decimal price, quantity, discount, total, final_price;   float float_price;   // Setting values   s21_from_float_to_decimal(19.99f, &price);   s21_from_int_to_decimal(100, &quantity);   s21_from_float_to_decimal(0.10f, &discount);   // 10% discount   // Calculate total: Price √ó Quantity   s21_mul(price, quantity, &total);   // Apply discount: Total - (Total √ó Discount)   s21_decimal discount_amount;   s21_mul(total, discount, &discount_amount);   s21_sub(total, discount_amount, &final_price);   // Conversion for display   s21_from_decimal_to_float(final_price, &float_price);   printf("Final price: $%.2f\n", float_price);   // $1,799.10 }
Example 2: Different Rounding Operations
void rounding_examples() {   s21_decimal value, rounded, floored, truncated;   int int_result;   s21_from_float_to_decimal(123.456789f, &value);   // Different rounding methods   s21_round(value, &rounded);   // 123   s21_floor(value, &floored);   // 123   s21_truncate(value, &truncated);   // 123   // For negative values   s21_decimal negative_value;   s21_from_float_to_decimal(-123.456789f, &negative_value);   s21_floor(negative_value, &floored);   // -124 }
