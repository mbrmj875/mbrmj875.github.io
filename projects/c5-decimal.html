أمثلة عملية على الاستخدام

مثال 1: الحسابات المالية الدقيقة
#include "s21_decimal.h"

void financial_calculations() {
    s21_decimal price, quantity, discount, total, final_price;
    float float_price;

    // تعيين القيم
    s21_from_float_to_decimal(19.99f, &price);
    s21_from_int_to_decimal(100, &quantity);
    s21_from_float_to_decimal(0.10f, &discount); // خصم 10%

    // حساب الإجمالي: السعر × الكمية
    s21_mul(price, quantity, &total);

    // تطبيق الخصم: الإجمالي - (الإجمالي × الخصم)
    s21_decimal discount_amount;
    s21_mul(total, discount, &discount_amount);
    s21_sub(total, discount_amount, &final_price);

    // التحويل للعرض
    s21_from_decimal_to_float(final_price, &float_price);
    printf("السعر النهائي: $%.2f\n", float_price); // $1,799.10
}

void rounding_examples() {
    s21_decimal value, rounded, floored, truncated;
    int int_result;

    s21_from_float_to_decimal(123.456789f, &value);

    // طرق التقريب المختلفة
    s21_round(value, &rounded); // 123
    s21_floor(value, &floored); // 123
    s21_truncate(value, &truncated); // 123

    // للقيم السالبة
    s21_decimal negative_value;
    s21_from_float_to_decimal(-123.456789f, &negative_value);
    s21_floor(negative_value, &floored); // -124
}
